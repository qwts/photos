<script type="text/babel">
const { Toast, TitleBar, Icon } = window.DS;

function App() {
  const [photos, setPhotos] = React.useState(window.OVERLOOK_PHOTOS);
  const [q, setQ] = React.useState("");
  const [zoom, setZoom] = React.useState(160);
  const [view, setView] = React.useState("grid");
  const [source, setSource] = React.useState("all");
  const [filters, setFilters] = React.useState({});
  const [filterOpen, setFilterOpen] = React.useState(false);
  const [selected, setSelected] = React.useState(new Set());
  const [lightboxId, setLightboxId] = React.useState(null);
  const [inspectorOpen, setInspectorOpen] = React.useState(false);
  const [importOpen, setImportOpen] = React.useState(false);
  const [dropped, setDropped] = React.useState(null);
  const [dragging, setDragging] = React.useState(false);
  const [settingsOpen, setSettingsOpen] = React.useState(false);
  const [keyMode, setKeyMode] = React.useState(null);
  const [connected, setConnected] = React.useState(true);
  const [sidebarCollapsed, setSidebarCollapsed] = React.useState(() => { try { return localStorage.getItem("overlook.sidebarCollapsed") === "1"; } catch (e) { return false; } });
  const toggleSidebar = () => setSidebarCollapsed((v) => { const n = !v; try { localStorage.setItem("overlook.sidebarCollapsed", n ? "1" : "0"); } catch (e) {} return n; });
  const [exportOpen, setExportOpen] = React.useState(false);
  const [toast, setToast] = React.useState(null);
  const [pendingCount, setPendingCount] = React.useState(362);
  const [lastBackupLabel, setLastBackupLabel] = React.useState("2H AGO");

  const visible = photos.filter((p) => {
    if (q && !(p.name + " " + p.place + " " + p.camera).toLowerCase().includes(q.toLowerCase())) return false;
    if (source === "favorites" && !p.favorite) return false;
    if (source === "offloaded" && p.status !== "offloaded") return false;
    if (source === "trash") return false;
    if (filters.favorites && !p.favorite) return false;
    if (filters.raw && !p.name.endsWith(".RAF")) return false;
    if (filters.offloaded && p.status !== "offloaded") return false;
    if (filters.local && p.status !== "local") return false;
    return true;
  });

  const idx = visible.findIndex((p) => p.id === lightboxId);
  const current = idx >= 0 ? visible[idx] : null;

  const toggle = (id) => setSelected((s) => { const n = new Set(s); n.has(id) ? n.delete(id) : n.add(id); return n; });
  const toggleFilter = (k) => setFilters((f) => ({ ...f, [k]: !f[k] }));
  const step = (d) => { if (idx >= 0 && visible.length) setLightboxId(visible[(idx + d + visible.length) % visible.length].id); };
  const toggleFav = () => {
    if (!current) return;
    setPhotos((ps) => ps.map((p) => p.id === current.id ? { ...p, favorite: !p.favorite } : p));
    setPendingCount((n) => n + 1);
  };
  const showToast = (t) => { setToast(t); setTimeout(() => setToast(null), 4000); };

  const dragDepth = React.useRef(0);
  const hasFiles = (e) => Array.from(e.dataTransfer && e.dataTransfer.types || []).includes("Files");
  const onDragEnter = (e) => { if (!hasFiles(e)) return; e.preventDefault(); dragDepth.current += 1; setDragging(true); };
  const onDragOver = (e) => { if (hasFiles(e)) e.preventDefault(); };
  const onDragLeave = (e) => { if (!hasFiles(e)) return; dragDepth.current -= 1; if (dragDepth.current <= 0) { dragDepth.current = 0; setDragging(false); } };
  const onDrop = (e) => {
    if (!hasFiles(e)) return;
    e.preventDefault();
    dragDepth.current = 0; setDragging(false);
    const files = Array.from(e.dataTransfer.files).filter((f) => /^image\//.test(f.type) || /\.(raf|arw|cr2|cr3|nef|dng|heic|jpe?g|png|tiff?)$/i.test(f.name));
    if (!files.length) { showToast({ tone: "amber", title: "Nothing to import", detail: "Drop photo files (RAW, JPEG, PNG, HEIC…)" }); return; }
    const bytes = files.reduce((s, f) => s + (f.size || 0), 0);
    const gb = (bytes / 1e9);
    setDropped({ count: files.length, detail: files.length + " file" + (files.length === 1 ? "" : "s") + (gb >= 0.05 ? " · " + gb.toFixed(1) + " GB" : "") });
    setImportOpen(true);
  };
  const runBackup = () => {
    if (!connected || pendingCount === 0) return;
    const n = pendingCount;
    showToast({ tone: "amber", title: "Backup started", detail: `Encrypting ${n} changed photo${n === 1 ? "" : "s"} → pCloud` });
    setTimeout(() => {
      setPendingCount(0);
      setLastBackupLabel("JUST NOW");
      showToast({ tone: "green", title: "Backup complete", detail: `${n} photo${n === 1 ? "" : "s"} encrypted and uploaded` });
    }, 1800);
  };

  React.useEffect(() => {
    const onKey = (e) => {
      if (lightboxId == null) {
        if ((e.metaKey || e.ctrlKey) && (e.key === "a" || e.key === "A")) {
          e.preventDefault();
          setSelected(new Set(visible.map((p) => p.id)));
        }
        if (e.key === "Escape" && selected.size) setSelected(new Set());
        return;
      }
      if (e.key === "Escape") setLightboxId(null);
      if (e.key === "ArrowLeft") step(-1);
      if (e.key === "ArrowRight") step(1);
      if (e.key === "i" || e.key === "I") setInspectorOpen((v) => !v);
    };
    window.addEventListener("keydown", onKey);
    return () => window.removeEventListener("keydown", onKey);
  });

  return (
    <div onDragEnter={onDragEnter} onDragOver={onDragOver} onDragLeave={onDragLeave} onDrop={onDrop} style={{ height: "100%", display: "flex", flexDirection: "column", position: "relative", overflow: "hidden", fontFamily: "var(--font-ui)", color: "var(--text-body)" }}>
      <TitleBar platform="mac" />
      <Toolbar q={q} setQ={setQ} zoom={zoom} setZoom={setZoom} view={view} setView={setView}
        filters={filters} toggleFilter={toggleFilter} filterOpen={filterOpen} setFilterOpen={setFilterOpen}
        onImport={() => setImportOpen(true)}
        onBackup={runBackup} pendingCount={pendingCount} connected={connected} />
      <div style={{ flex: 1, minHeight: 0, display: "flex" }}>
        <Sidebar source={source} setSource={setSource} total={204318} onSettings={() => setSettingsOpen(true)} connected={connected} collapsed={sidebarCollapsed} onToggle={toggleSidebar} />
        <LibraryGrid photos={visible} zoom={zoom} view={view} selected={selected} onToggle={toggle}
          onOpen={(id) => { setLightboxId(id); }} onClearSelection={() => setSelected(new Set())}
          onExport={() => setExportOpen(true)} />
        {inspectorOpen && !current ? <Inspector photo={visible[0]} /> : null}
      </div>
      <StatusBar count={204318} pendingCount={pendingCount} lastBackupLabel={lastBackupLabel} connected={connected} />
      {current ? (
        <div style={{ position: "absolute", inset: 0, zIndex: 50, display: "flex" }}>
          <Lightbox photo={current} onClose={() => setLightboxId(null)} onPrev={() => step(-1)} onNext={() => step(1)}
            inspectorOpen={inspectorOpen} setInspectorOpen={setInspectorOpen} onToggleFavorite={toggleFav}
            onExport={() => setExportOpen(true)} />
          {inspectorOpen ? <Inspector photo={current} /> : null}
        </div>
      ) : null}
      <ImportDialog open={importOpen} dropped={dropped} onClose={() => { setImportOpen(false); setDropped(null); }}
        onDone={() => showToast({ tone: "green", title: "Import complete", detail: "Photos encrypted · thumbnails ready" })} />
      {dragging ? (
        <div style={{ position: "absolute", inset: 0, zIndex: 400, display: "flex", alignItems: "center", justifyContent: "center", background: "color-mix(in srgb, var(--gray-0) 72%, transparent)", backdropFilter: "blur(2px)", pointerEvents: "none" }}>
          <div style={{ display: "flex", flexDirection: "column", alignItems: "center", gap: 12, border: "2px dashed var(--accent-cyan)", borderRadius: "var(--radius-3)", padding: "40px 64px", background: "var(--gray-1)" }}>
            <Icon name="image-down" size={40} color="var(--accent-cyan)" />
            <div style={{ fontSize: "var(--text-lg)", fontWeight: 600, color: "var(--text-body)" }}>Drop photos to import</div>
            <div style={{ font: "var(--type-meta)", color: "var(--text-faint)" }}>Encrypted on this device · RAW, JPEG, PNG, HEIC</div>
          </div>
        </div>
      ) : null}
      <SettingsDialog open={settingsOpen} onClose={() => setSettingsOpen(false)} onKeyAction={(m) => setKeyMode(m)} connected={connected} setConnected={setConnected} />
      <KeyDialog open={keyMode != null} mode={keyMode} onClose={() => setKeyMode(null)} onToast={showToast} />
      <ExportDialog open={exportOpen} onClose={() => setExportOpen(false)} count={current ? 1 : Math.max(1, selected.size)} />
      {toast ? (
        <div style={{ position: "absolute", right: 16, bottom: 42, zIndex: 300 }}>
          <Toast tone={toast.tone} title={toast.title} detail={toast.detail} />
        </div>
      ) : null}
    </div>
  );
}
ReactDOM.createRoot(document.getElementById("root")).render(<App />);
</script>
