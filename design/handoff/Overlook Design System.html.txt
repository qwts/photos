      home=document.getElementById('home'), ttl=document.getElementById('ttl'), sub=document.getElementById('sub'),
      openNew=document.getElementById('openNew'), ovCards=document.getElementById('ovCards');

function show(card, btn){
  document.querySelectorAll('.item.active').forEach(e=>e.classList.remove('active'));
  if (btn) btn.classList.add('active');
  ttl.textContent = card.name;
  sub.textContent = card.sub || '';
  if (card.home){
    home.classList.remove('hidden'); wrap.classList.add('hidden'); openNew.classList.add('hidden');
    frame.src='about:blank'; return;
  }
  home.classList.add('hidden'); wrap.classList.remove('hidden'); openNew.classList.remove('hidden');
  openNew.href = card.src;
  if (card.full){ frame.className='full'; wrap.classList.add('full'); frame.style.width=''; frame.style.height=''; }
  else { frame.className=''; wrap.classList.remove('full'); frame.style.width=card.w+'px'; frame.style.height=card.h+'px'; }
  frame.src = card.src;
}
